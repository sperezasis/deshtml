    ```bash
    set -e
    SOURCE=/Users/sperezasis/work/caseproof/DOCUMENTATION-SYSTEM.md
    PALETTE=skill/design/palette.css
    fail=0
    while IFS= read -r line; do
      # Match lines like:   --black: #2C2C2E;      /* comment */
      # Strip leading/trailing whitespace, keep var: value;
      var_decl="$(echo "$line" | sed -E 's/^[[:space:]]*//; s/[[:space:]]*\/\*.*\*\///; s/[[:space:]]*$//')"

      var_only="$(echo "$var_decl" | sed -E 's/;$//')"
      if ! grep -qF "$var_only;" "$SOURCE"; then
        echo "DRIFT: '$var_only;' not found verbatim in $SOURCE" >&2

      fi
    ```
